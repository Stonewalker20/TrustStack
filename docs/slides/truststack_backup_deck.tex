\begin{frame}[shrink=8]{1. Introduction and Problem Setting}
  \vspace*{-0.28cm}
  \begin{columns}[T,totalwidth=\textwidth]
    \column{0.58\textwidth}
    \slidelead{TrustStack starts from a simple problem: a fluent answer is not enough when a user must decide whether an AI output is safe, supported, and auditable before using it.}
    \begin{block}{Research question}
      How can a user judge whether an AI answer is trustworthy when conventional chat interfaces expose fluency but hide evidence quality, contradiction risk, and review conditions?
    \end{block}
    \begin{block}{Project thesis}
      TrustStack reframes trust as an evidence-grounded evaluation task. It turns trust into a structured review artifact built from evidence, scoring, risk signals, and explanation.
    \end{block}
    \column{0.38\textwidth}
    \statcard{Core task}{Grounded trust review}
    \vspace{0.45em}
    \statcard{Primary risk}{Unsupported confidence}
    \vspace{0.45em}
    \statcard{Decision basis}{Evidence + diagnostics}
  \end{columns}
\end{frame}

\begin{frame}{2. Related Work and Contribution}
  \slidelead{TrustStack sits between retrieval-augmented generation, hallucination analysis, and structured evaluation frameworks. The contribution is not a new base model. It is a new evidence-first evaluation layer around existing generation pipelines.}
  \begin{columns}[T,totalwidth=\textwidth]
    \column{0.32\textwidth}
    \begin{block}{RAG systems}
      Model outputs should be conditioned on retrieved evidence rather than unconstrained recall.
    \end{block}
    \column{0.32\textwidth}
    \begin{block}{Reliability research}
      Plausible answers can still be unsupported, unsafe, or operationally misleading.
    \end{block}
    \column{0.32\textwidth}
    \begin{block}{Evaluation frameworks}
      Reproducible prompts, dimensions, judgments, and exportable artifacts matter.
    \end{block}
  \end{columns}
  \vspace{0.6em}
  \begin{block}{TrustStack contribution}
      TrustStack integrates these threads into one local-first workflow centered on answer inspection rather than answer appearance.
  \end{block}
\end{frame}

\begin{frame}{3. Data}
  \slidelead{The evaluation uses two data families: live evidence uploaded by the user and benchmark-oriented corpora used to stress the system under reproducible conditions.}
  \begin{columns}[T,totalwidth=\textwidth]
    \column{0.48\textwidth}
    \begin{block}{Live evidence path}
      \small
      \begin{itemize}
        \item PDF, DOCX, TXT, and Markdown uploads
        \item Parsed, chunked, indexed, and stored in MongoDB
        \item Used for real-time evidence-grounded question answering
      \end{itemize}
    \end{block}
    \column{0.48\textwidth}
    \begin{block}{Benchmark path}
      \small
      \begin{itemize}
        \item Seven synthetic stress-test packets
        \item Checked-in public benchmark subsets: SciFact and HotpotQA
        \item Same evaluation protocol applied across both paths
      \end{itemize}
    \end{block}
  \end{columns}
  \vspace{0.6em}
  \begin{center}
    \metricpill{tsteal}{User uploads}\hspace{0.5em}
    \metricpill{tsblue}{Synthetic stress tests}\hspace{0.5em}
    \metricpill{tswarn}{External benchmarks}
  \end{center}
\end{frame}

\begin{frame}{4. Methods and System Architecture}
  \slidelead{TrustStack is implemented as a five-layer pipeline so each stage can be validated independently and each failure can be localized.}
  \begin{block}{Pipeline}
    \centering
    \metricpill{tsblue}{1. Ingest}\hspace{0.4em}
    \metricpill{tsteal}{2. Retrieve}\hspace{0.4em}
    \metricpill{ougold}{3. Generate}\hspace{0.4em}
    \metricpill{tsblue}{4. Evaluate}\hspace{0.4em}
    \metricpill{tsteal}{5. Review / Export}
  \end{block}
  \vspace{0.6em}
  \begin{columns}[T,totalwidth=\textwidth]
    \column{0.48\textwidth}
    \begin{block}{Implementation}
      \small
      \begin{itemize}
        \item MongoDB-backed documents, chunks, and run history
        \item Retrieval via vector or lexical fallback
        \item Grounded answer generation with deterministic fallback mode
      \end{itemize}
    \end{block}
    \column{0.48\textwidth}
    \begin{block}{Verification stack}
      \small
      \begin{itemize}
        \item Backend unit tests
        \item Real Mongo integration tests
        \item Browser E2E tests
        \item Served-stack smoke verification
      \end{itemize}
    \end{block}
  \end{columns}
\end{frame}

\begin{frame}{5. TrustStack Evaluation Standard v2.0}
  \slidelead{The TrustStack Evaluation Standard converts one answer into an evidence-first review packet rather than a single opaque score.}
  \begin{block}{Ten scored dimensions}
    Retrieval relevance, evidence sufficiency, citation traceability, claim support, contradiction risk, completeness, abstention behavior, answer discipline, safety risk, and calibration.
  \end{block}
  \vspace{0.5em}
  \begin{columns}[T,totalwidth=\textwidth]
    \column{0.32\textwidth}
    \statcard{Verdicts}{Pass / Review / Fail}
    \column{0.32\textwidth}
    \statcard{Diagnostics}{Claims + checks + flags}
    \column{0.32\textwidth}
    \statcard{Output}{Reproducible report artifact}
  \end{columns}
  \vspace{0.6em}
    \begin{block}{Why this matters}
      The system shows \emph{why} confidence moved, what evidence was missing, and when human review is required.
    \end{block}
\end{frame}

\begin{frame}[shrink=8]{6. Synthetic Benchmark Findings}
  \slidelead{The synthetic benchmark tests whether TrustStack reacts sensibly when the evidence environment changes under controlled conditions.}
  \begin{columns}[T,totalwidth=\textwidth]
    \column{0.52\textwidth}
    \begin{block}{Synthetic evaluation design}
      Seven corpus conditions stress aligned evidence, contradiction, sparse evidence, lexical drift, unsafe procedural advice, and consensus-style multi-document overlap.
    \end{block}
    \begin{block}{Interpretation}
      The benchmark does not claim absolute trustworthiness. It tests whether scores move in the expected direction when evidence quality changes.
    \end{block}
    \column{0.44\textwidth}
    \statcard{Best synthetic score}{76.44 / 100}
    \vspace{0.45em}
    \statcard{Worst synthetic score}{59.97 / 100}
    \vspace{0.45em}
    \statcard{Weakest category}{Grounding + retrieval}
  \end{columns}
\end{frame}

\begin{frame}[shrink=6]{7. External Benchmark Findings}
  \slidelead{TrustStack was also evaluated on checked-in public benchmark subsets from SciFact and HotpotQA, extending the evaluation beyond internally authored stress corpora.}
  \begin{columns}[T,totalwidth=\textwidth]
    \column{0.48\textwidth}
    \begin{block}{What this demonstrates}
      The paper no longer relies only on internally authored synthetic corpora. External benchmark slices show that the same score and verdict logic can operate on public tasks while preserving reproducible diagnostics.
    \end{block}
    \begin{block}{Interpretation}
      Citation alignment is already strong, and the remaining spread between trust score and task metric identifies the clearest next optimization target for retrieval and semantic answer quality.
    \end{block}
    \column{0.48\textwidth}
    \statcard{Aggregate TrustStack score}{63.09 / 100}
    \vspace{0.45em}
    \statcard{Task metric}{0.141}
    \vspace{0.45em}
    \statcard{Citation alignment}{83.8\%}
  \end{columns}
\end{frame}

\begin{frame}[shrink=6]{8. Validity and Next-Stage Readiness}
  \slidelead{A defensible AI project should define both what has been demonstrated and where the platform is ready to scale next.}
  \begin{columns}[T,totalwidth=\textwidth]
    \column{0.49\textwidth}
    \begin{block}{Current baseline}
      \small
      \begin{itemize}
        \item Local lexical retrieval provides a stable, reproducible starting point.
        \item Extractive fallback generation keeps the runtime transparent and auditable.
        \item Human analyst studies are a clear next validation stage.
      \end{itemize}
    \end{block}
    \column{0.49\textwidth}
    \begin{block}{Why the platform is credible}
      \small
      \begin{itemize}
        \item The evaluation is explicit, measured, and tied to concrete next experiments.
        \item The paper now includes real external benchmarks, not just synthetic data.
        \item The evaluation and export stack is reproducible end to end.
      \end{itemize}
    \end{block}
  \end{columns}
\end{frame}

\begin{frame}[shrink=4]{9. Conclusion}
  \slidelead{TrustStack contributes a reusable workflow for grounded evaluation, benchmarking, and reporting.}
  \begin{block}{Main contribution}
    TrustStack is valuable not because it makes AI look more persuasive, but because it makes trust inspectable. It connects live evidence review, a formal evaluation standard, synthetic benchmarking, public-benchmark evaluation, and exportable IEEE-ready artifacts into one coherent system.
  \end{block}
